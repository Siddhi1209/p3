
<?php
$basePath = "/opt/hpws/apache/cgi-bin/trials/bip/data/TESTING/";
$selectedDate = $_GET['date'] ?? date("Ymd");
$filePath = $basePath . "invalid_neft_" . $selectedDate . ".txt";

$total=0; $success=0; $failed=0;
$dataRows=[];

if (file_exists($filePath)) {
    $handle=fopen($filePath,"r");
    $first=true;

    while(($line=fgets($handle))!==false){
        $line=trim($line);

        if($first){$first=false; continue;}
        if($line=="no Invalid UTR" || empty($line)) continue;

        $data=explode(",",$line);
        $dataRows[]=$data;

        $total++;
        if(strtoupper($data[2]??"")=="SUCCESS") $success++;
        if(strtoupper($data[2]??"")=="FAILED") $failed++;
    }
    fclose($handle);
}

<script>
var table = $('#neftTable').DataTable({
    scrollX:true,
    pageLength:25
});

// FILTER FUNCTION
function filterTable(type){
    if(type=="ALL"){
        table.rows().nodes().to$().show();
    } else {
        table.rows().nodes().to$().each(function(){
            var status=$(this).data('status');
            if(status==type) $(this).show();
            else $(this).hide();
        });
    }
}

// TIME
function updateTime(){
    let now=new Date();
    let m=["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"];
    document.getElementById("time").innerHTML=
        now.getDate()+" "+m[now.getMonth()]+" "+
        now.getHours().toString().padStart(2,'0')+":"+
        now.getMinutes().toString().padStart(2,'0');
}
setInterval(updateTime,1000);
updateTime();
