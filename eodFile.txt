
<?php
date_default_timezone_set("Asia/Kolkata");

/* ================= CONFIG ================= */
$basePath = "/opt/hpws/apache/cgi-bin/trials/bip/data/TESTING/";

/* =============== DATE HANDLING =============== */
$selectedDateInput = isset($_GET['date']) ? $_GET['date'] : date("Y-m-d");
$selectedDate = str_replace("-", "", $selectedDateInput);

$filePath = $basePath . "invalid_neft_" . $selectedDate . ".txt";

/* =============== READ FILE =============== */
$dataRows = [];
$noDataFlag = false;

if (file_exists($filePath)) {

    $handle = fopen($filePath, "r");

    while (($line = fgets($handle)) !== false) {

        $line = trim($line);
        $line = str_replace("\r", "", $line);

        if ($line == "") continue;

        if (strtolower($line) == "no invalid utr") {
            $noDataFlag = true;
            break;
        }

        $data = explode(",", $line);

        /* FORCE EXACTLY 13 COLUMNS */
        $data = array_slice($data, 0, 13);
        $data = array_pad($data, 13, "");

        $dataRows[] = $data;
    }

    fclose($handle);
}

/* PRINT MESSAGE ROW (IMPORTANT FIX) */
function printMessageRow($msg) {
    echo "<tr>";
    for ($i = 0; $i < 13; $i++) {
        if ($i == 0) {
            echo "<td style='font-weight:bold;'>$msg</td>";
        } else {
            echo "<td></td>";
        }
    }
    echo "</tr>";
}
