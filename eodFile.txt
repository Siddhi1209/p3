<?php
########################################################
# DATE LIST VIEW
########################################################
if (empty($selectedDate)) {
?>
    <div class="card">
        <h4>Select Date</h4>
        <?php
        foreach ($files as $file) {
            $date = str_replace(["file_", ".txt"], "", basename($file));
            echo "<a class='date-link' href='?date=$date'>$date</a>";
        }
        ?>

<?php
}
########################################################
# DATA TABLE VIEW
########################################################
else {

    $filePath = $basePath . "file_" . $selectedDate . ".txt";
?>

    <div class="card">
        <a href="?" class="btn btn-default">⬅ Back</a>
        <h4 style="margin-top:10px;">UTR Data for Date: <b><?php echo $selectedDate; ?></b></h4>

        <table id="neftTable" class="display table table-bordered table-striped">
            <thead>
                <tr>
                    <th>UTR</th>
                    <th>IN_OUT</th>
                    <th>STATUS</th>
                    <th>TRAN_DATE</th>
                    <th>AMOUNT</th>
                    <th>ACCOUNT_NO</th>
                    <th>BENEF_ACCT_NO</th>
                    <th>JRNL_NO</th>
                    <th>BRANCH</th>
                </tr>
            </thead>
            <tbody>

<?php
    if (!file_exists($filePath)) {
        echo "<tr><td colspan='9'>File not found</td></tr>";
    } else {
        $handle = fopen($filePath, "r");
        $first = true;

        while (($line = fgets($handle)) !== false) {
            $line = trim($line);

            if ($line == "no Invalid UTR") {
                echo "<tr><td colspan='9'>No Invalid UTR Found</td></tr>";
                break;
            }

            if ($first) {
                $first = false;
                continue;
            }

            $data = explode(",", $line);

            // Row highlight
            $rowClass = "";
            if (isset($data[2])) {
                if (strtoupper($data[2]) == "SUCCESS") {
                    $rowClass = "success-row";
                } elseif (strtoupper($data[2]) == "FAILED") {
                    $rowClass = "failed-row";
                }
            }

            echo "<tr class='$rowClass'>";
            foreach ($data as $value) {
                echo "<td>" . htmlspecialchars($value) . "</td>";
            }
            echo "</tr>";
        }

        fclose($handle);
    }
?>

            </tbody>
        </table>
    </div>

<?php } ?>
