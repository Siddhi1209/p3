<?php
if (!file_exists($filePath)) {
    echo "<tr><td colspan='13'>File not found</td></tr>";
} else {

    $handle = fopen($filePath, "r");
    $first = true;

    while (($line = fgets($handle)) !== false) {

        $line = trim($line);

        if ($line == "no Invalid UTR") {
            echo "<tr><td colspan='13'>No Invalid UTR Found</td></tr>";
            break;
        }

        // skip header
        if ($first) {
            $first = false;
            continue;
        }

        $data = explode(",", $line);

        echo "<tr>";

        // Ensure 13 columns always (avoid misalignment)
        for ($i = 0; $i < 13; $i++) {
            echo "<td>" . htmlspecialchars($data[$i] ?? "") . "</td>";
        }

        echo "</tr>";
    }

    fclose($handle);
}
