
<?php
$basePath = "/opt/hpws/apache/cgi-bin/trials/bip/data/TESTING/";
$selectedDate = $_GET['date'] ?? date("Ymd");
$filePath = $basePath . "invalid_neft_" . $selectedDate . ".txt";

$dataRows = [];
$noDataFlag = false;

if (file_exists($filePath)) {
    $handle = fopen($filePath, "r");
    $first = true;

    while (($line = fgets($handle)) !== false) {
        $line = trim($line);

        if (strtolower($line) == "no invalid utr") {
            $noDataFlag = true;
            break;
        }

        if ($first) {
            $first = false;
            continue;
        }

        if (!empty($line)) {
            $data = array_map('trim', explode(",", $line));
            $dataRows[] = $data;
        }
    }

    fclose($handle);
}

$colCount = 13;

// CASE 1: No Invalid UTR
if ($noDataFlag) {
    echo "<tr>";

        if ($i == 0) {
            echo "<td><b>No Invalid UTR</b></td>";
        } else {
            echo "<td></td>";
        }
    }
    echo "</tr>";
}

// CASE 2: Data present
elseif (!empty($dataRows)) {

    foreach ($dataRows as $row) {

        $status = strtoupper($row[2] ?? "");
        $class = "";

        if ($status == "SUCCESS") $class = "success-row";
        if ($status == "FAILED") $class = "failed-row";

        echo "<tr>";

        for ($i = 0; $i < $colCount; $i++) {
            echo "<td>" . htmlspecialchars($row[$i] ?? "") . "</td>";
        }

        echo "</tr>";
    }
}

// CASE 3: Empty
else {
