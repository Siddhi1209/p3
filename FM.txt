3. Before </body> tag
Add this JavaScript at the end of the <body> section, just before </body>.

html
Copy
Edit
    <script>
        function checkTaskStatus() {
            $.ajax({
                url: 'monitorExternalFile1.php',
                method: 'GET',
                success: function(response) {
                    if (response.trim() === 'start') {
                        $("#taskPopup").fadeIn();
                        window.location.href = 'legends.php';
                    } else {
                        $("#taskPopup").fadeOut();
                    }
                }
            });
        }

        $(document).ready(function() {
            checkTaskStatus();
            setInterval(checkTaskStatus, 5000);
            $("#closePopup").click(function() {
                $("#taskPopup").fadeOut();
            });
        });
    </script>
