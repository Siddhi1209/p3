<?php

if(file_exists($file))
{

    $fp = fopen($file,"r");

    $firstLine = true;
    $count = 0;

    while(($line = fgets($fp)) !== false)
    {

        $line = trim($line);

        if($line == "")
            continue;

        /* Skip CSV Header */

        if($firstLine)
        {
            $firstLine = false;
            continue;
        }

        $data = explode(",",$line);

        echo "<tr>";

        /* LOG TYPE */

        $logType = trim($data[0]);

        if($logType == "NOPUSH")
        {
            echo "<td class='nopush'>".$logType."</td>";
        }
        elseif($logType == "FAILED")
        {
            echo "<td class='failed'>".$logType."</td>";
        }
        elseif($logType == "IGNORED")
        {
            echo "<td class='ignored'>".$logType."</td>";
        }
        else
        {
            echo "<td>".$logType."</td>";
        }

        /* Remaining Columns */

        for($i=1;$i<7;$i++)
        {

            if(isset($data[$i]))
            {
                echo "<td>".htmlspecialchars($data[$i])."</td>";
            }
            else
            {
                echo "<td>&nbsp;</td>";
            }

        }

        echo "</tr>";

        $count++;

    }

    fclose($fp);

    if($count==0)
    {

        echo "<tr>";
        echo "<td colspan='7' class='message'>";
        echo "No Pending FTP Records Found";
        echo "</td>";
        echo "</tr>";

    }

}
else
{

    echo "<tr>";
    echo "<td colspan='7' class='message'>";
    echo "Report File Not Available";
    echo "</td>";
    echo "</tr>";

}
