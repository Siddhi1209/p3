#!/bin/ksh
pgm=`basename $0`
rm -f $sysout/$pgm.log
DB_QUERY() {
        queryResult=`sqlplus -s / << YSPID
        set termout off
        set veri off
        set head off
        set feed off
        set pages 0
        set null #
        $@;
        show sqlcode;
        exit;
 YSPID`
        export sqlCode=`echo "$queryResult" | grep sqlcode | awk '{print $2}'`
        export queryResult=`echo "$queryResult" | grep -v sqlcode`
        if [ $sqlCode -ne 0 ];
        then
         DISP_ERROR echo "Problem with Database connection or error in sql command SQLCODE= " $sqlCode
         exit 5
        fi
 }
 DISP_ERROR()
 {
        if [ "$AUTOSYS_MODE" = "Y" ];
        then
                banner ERROR
                echo "\n"
                echo "Error in `hostind` at `date` $1" >> $sysout/zerobyte.log
                exit 10
        else
                banner ERROR
                echo "\n"
                echo "Error in `hostind` at `date` $1" >> $sysout/$pgm.log
                exit 10
        fi
 }

 DB_QUERY "select dqptype from hostind where status='R'"
 export $queryResult

 for i in `echo $queryResult`
 do
         export $i
         if [ "$i" = "$BANCS_DQPTYPE" ];then
              echo $i
         else
              echo $i
              fileTransfer.sh 6 $data/temp $data/temp/ BORTOGL_??? $LOGNAME $i
              if [ $? -eq 0 ] ;
                then
                echo "File Transfer Successful" | tee -a $sysout/$pgm.log
              else
                 if [ "$AUTOSYS_MODE" = "Y" ] ; then
                        echo "File Transfer Fail for BORTOGL_??? " >> $sysout/zerobyte.log
                        $sh/checkrc.sh 14 $0
                        break 1;
                 else
                        echo "File Transfer Failed: FILE NAME:BORTOGL_??? " | tee -a $sysout/$pgm.log
                        echo "IS IT OK TO CONTINUE [Y/N] :\c" | tee -a $sysout/$pgm.log
                        read reply
                        case $reply in
                             yes|YES) break;;
                             no|NO) exit;
                        esac
                 fi
              fi
         fi
 done
cd $data/temp
rm -f processed_accounts_BR0501.txt.gz
total_count=`ls BORTOGL_??? |wc -l`
max=$(ls BORTOGL_[0-9][0-9][0-9] | sed 's/[^0-9]//g' | sort -n | tail -1)
echo $total_count >> $sysout/$pgm.log
count=0
while [ $count -lt $max ];
do
 count=`expr $count + 1`
 aaa=`printf "%03d" $count`
 f=BORTOGL_${aaa}
 if [ -f "$f" ]; then
    if [ "$AUTOSYS_MODE" = "Y" ]; then
     echo "Processing $f ..." >> $sysout/zerobyte.log
    else
     echo "Processing $f ..." >> $sysout/$pgm.log
    fi
 awk '{print substr($0,1,19)" " substr($0,25,2)}' "$f" >> processed_accounts_BR0501.txt
 else
    if [ "$AUTOSYS_MODE" = "Y" ]; then
     echo "Skipping missed file $f" >> $sysout/zerobyte.log
    else
     echo "Skipping missed file $f" >> $sysout/$pgm.log
    fi
 fi
done
Total=$(sed '1d;$d' processed_accounts_BR0501.txt | wc -l)
echo "Total processed records in file is : $Total" >> $sysout/$pgm.log
gzip processed_accounts_BR0501.txt
