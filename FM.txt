        if($firstLine)

            continue;
        }

        $data=explode(",",$line);

        echo "<tr>";

        for($i=0;$i<7;$i++)
        {
            echo "<td>".htmlspecialchars(isset($data[$i]) ? $data[$i] : "")."</td>";
        }

        echo "</tr>";

        $records++;
    }

    fclose($fp);

    echo "</table>";

    if($records==0)
